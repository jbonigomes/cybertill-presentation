function myFunc() {
  return 'test';
}

Don't

              
 function myFunc() {
   return 'test';

 }

Do

              
function myFunc(input) {
  var output = input;

  if (output.length) {
    // ...
  }

  for (var i = 0; i < output.length; i++) {
    // ...
  }

  return output;
}

Don't

              
function myFunc(input) {
  var output = input;
  if (output.length) {
    // ...
  }
  for (var i = 0; i < output.length; i++) {
    // ...
  }
  return output;
}
